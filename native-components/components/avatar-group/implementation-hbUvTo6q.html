iOS Usage Example 


                                                        
                                                        
                                                            // example Swift code using the component
                                                        
                                                        AvatarGroup(
                                                            faces: authors, 
                                                            size: .small
                                                        )
                                                        
                                                        //Avatar Group with button
                                                        AvatarGroup(
                                                            faces: Array(self.sampleFaces),
                                                            size: .medium,
                                                            additionalButton: AvatarGroup.ButtonData(icon: .plus, action: { })
                                                        )

 

 

 

Android Usage Example 


                                                        
                                                        
                                                            // example Kotlin code using the component
                                                        
                                                        // Avatar Group with Overflow
                                                        TAAtlasAvatarGroup(
                                                            avatarIconSize = AtlasAvatarSize.MEDIUM,
                                                            avatarsDisplayed = sampleAvatarImages(5),
                                                            totalNumberOfAvatars = 8
                                                        )
                                                        
                                                        
                                                        // Avatar Group with CTA 
                                                        TAAtlasAvatarGroup(
                                                            avatarIconSize = AtlasAvatarSize.MEDIUM,
                                                            avatarsDisplayed = sampleAvatarImages(6),
                                                            totalNumberOfAvatars = 5,
                                                            cta = CTAButtonData(accessibilityLabel = "Add more", onClick = {})
                                                        )
                                                        
                                                        // Avatar Group with Overflow and CTA
                                                        TAAtlasAvatarGroup(
                                                            avatarIconSize = AtlasAvatarSize.MEDIUM,
                                                            avatarsDisplayed = sampleAvatarImages(2),
                                                            totalNumberOfAvatars = 5,
                                                            cta = CTAButtonData(accessibilityLabel = "Add more", onClick = {})
                                                        )
