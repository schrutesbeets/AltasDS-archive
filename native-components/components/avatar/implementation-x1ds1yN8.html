iOS Usage Example 


                                                        
                                                        
                                                            // example Swift code using the component
                                                        
                                                        //Standard
                                                        Avatar(
                                                          imageSource: $0,
                                                          variant: .standard(size: .medium) 
                                                         )
                                                        
                                                        //With badge - extraLarge only
                                                        Avatar(
                                                          imageSource: store.avatar, 
                                                          variant: .badge(type: .editPhoto()),
                                                          tapAction: photoAction
                                                        )

 

 

 

Android Usage Example 


                                                        
                                                        
                                                            // example Kotlin code using the component
                                                        
                                                        //Standard
                                                        TAAtlasAvatar(
                                                            avatarSize = AtlasAvatarSize.MEDIUM,
                                                            onClick = {}
                                                        )
                                                        
                                                        //With badge - extraLarge only
                                                        TAAtlasAvatar(
                                                            avatarSize = AtlasAvatarSize.X_LARGE,
                                                            imageSource = null,
                                                            badgeIcon = AtlasBadgeIcon.EDIT_PHOTO,
                                                            onClick = {}
                                                        )
